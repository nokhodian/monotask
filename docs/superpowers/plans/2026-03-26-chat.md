# P2P Chat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add per-space persistent P2P chat with `@mention` and `#ref` autocomplete, rendered in a floating panel; chat docs sync via the existing automerge board sync pipeline.

**Architecture:** Each space gets a dedicated automerge doc keyed `{space_id}-chat`, stored in the `boards` table with `is_system=1` (so it's invisible to the boards grid). Chat messages are appended to a CRDT list. A `card_search_index` SQLite table enables cheap card title lookups for autocomplete. Synced identically to boards via `trigger_board_sync`.

**Prerequisite:** Rich Identity plan (2026-03-26-rich-identity.md) must be completed first — this plan uses `peer_profiles` and `MemberView.color_accent`.

**Tech Stack:** Rust (automerge, rusqlite, Tauri v2), vanilla JS/HTML

```rust
pub fn run_migrations_v3(conn: &Connection) -> Result<()> {
    let version: i64 = conn
        .query_row("PRAGMA user_version", [], |r| r.get(0))
        .unwrap_or(0);
    if version >= 3 {
        return Ok(());
    }
    conn.execute_batch("
        BEGIN;
        CREATE TABLE IF NOT EXISTS card_search_index (
            card_id     TEXT PRIMARY KEY,
            board_id    TEXT NOT NULL,
            space_id    TEXT NOT NULL,
            title       TEXT NOT NULL,
            column_name TEXT NOT NULL
        );
        CREATE INDEX IF NOT EXISTS idx_card_search_space
            ON card_search_index (space_id, title);
        COMMIT;
    ")?;
    conn.execute_batch("PRAGMA user_version = 3")?;
    Ok(())
}
```

```rust
pub fn list_messages(doc: &AutoCommit, limit: usize, before_ts: Option<u64>) -> crate::Result<Vec<ChatMessage>> {
    let (_, list_id) = doc.get(automerge::ROOT, "messages")?
        .ok_or_else(|| crate::Error::InvalidDocument("chat missing messages list".into()))?;
    let len = doc.length(&list_id);
    let mut msgs: Vec<ChatMessage> = Vec::new();

    for i in 0..len {
        let (_, entry) = match doc.get(&list_id, i)? {
            Some(v) => v,
            None => continue,
        };
        let id = crate::get_string(doc, &entry, "id")?.unwrap_or_default();
        let author = crate::get_string(doc, &entry, "author")?.unwrap_or_default();
        let text = crate::get_string(doc, &entry, "text")?.unwrap_or_default();
        let created_at = match doc.get(&entry, "created_at")? {
            Some((automerge::Value::Scalar(s), _)) => {
                match s.as_ref() {
                    automerge::ScalarValue::Uint(n) => *n,
                    automerge::ScalarValue::Int(n) => *n as u64,
                    _ => 0,
                }
            }
            _ => 0,
        };

        if let Some(before) = before_ts {
            if created_at >= before { continue; }
        }

        // Refs
        let refs = if let Some((_, refs_list)) = doc.get(&entry, "refs")? {
            let rlen = doc.length(&refs_list);
            (0..rlen).filter_map(|j| {
                let (_, ref_entry) = doc.get(&refs_list, j).ok()??;
                let kind = crate::get_string(doc, &ref_entry, "kind").ok()?.unwrap_or_default();
                let ref_id = crate::get_string(doc, &ref_entry, "id").ok()?.unwrap_or_default();
                let label = crate::get_string(doc, &ref_entry, "label").ok()?.unwrap_or_default();
                Some(ChatRef { kind, id: ref_id, label })
            }).collect()
        } else { vec![] };

        msgs.push(ChatMessage { id, author, text, created_at, refs });
    }

    // Newest first, then limit
    msgs.sort_by(|a, b| b.created_at.cmp(&a.created_at));
    msgs.truncate(limit);
    Ok(msgs)
}
```

```rust
#[tauri::command]
fn send_chat_message_cmd(
    space_id: String,
    text: String,
    refs: Vec<ChatRefInput>,
    state: tauri::State<AppState>,
) -> Result<(), String> {
    let pubkey = state.identity.public_key_hex();
    let storage = state.storage.lock().map_err(|e| e.to_string())?;
    let mut doc = get_or_create_chat_doc(&storage, &space_id)?;
    let msg = kanban_core::chat::ChatMessage {
        id: uuid::Uuid::new_v4().to_string(),
        author: pubkey,
        text,
        created_at: std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .unwrap_or_default()
            .as_secs(),
        refs: refs.into_iter().map(|r| kanban_core::chat::ChatRef {
            kind: r.kind, id: r.id, label: r.label,
        }).collect(),
    };
    kanban_core::chat::append_message(&mut doc, &msg).map_err(|e| e.to_string())?;
    storage.save_board(&format!("{space_id}-chat"), &doc, true).map_err(|e| e.to_string())?;
    drop(storage);
    // Trigger sync
    let net = state.net.lock().map_err(|e| e.to_string())?;
    if let Some(ref handle) = *net {
        handle.trigger_sync_sync(format!("{space_id}-chat"));
    }
    Ok(())
}
```

```rust
    // Build author → member profile lookup

    let member_map: std::collections::HashMap<String, kanban_core::space::Member> = space

            id: m.id,
            author: m.author,
            display_name: member.and_then(|mem| mem.display_name.clone()),
            avatar_b64: member.and_then(|mem| mem.avatar_blob.as_ref().map(|b| {

            refs: m.refs.into_iter().map(|r| ChatRefView { kind: r.kind, id: r.id, label: r.label }).collect(),
```

```rust
#[derive(serde::Serialize)]
struct MentionSuggestion {
    kind: String,     // "member" | "card" | "board"
    id: String,
    label: String,
    sublabel: Option<String>,   // role for members, column for cards
    avatar_b64: Option<String>,
    color_accent: Option<String>,
}

#[tauri::command]
fn get_mention_suggestions_cmd(
    space_id: String,
    query: String,
    kind: String,   // "all" | "member" | "card" | "board"
    state: tauri::State<AppState>,
) -> Result<Vec<MentionSuggestion>, String> {
    let storage = state.storage.lock().map_err(|e| e.to_string())?;
    let q = query.to_lowercase();
    let mut results: Vec<MentionSuggestion> = Vec::new();

    if kind == "all" || kind == "member" {
        if let Ok(space) = kanban_storage::space::get_space(storage.conn(), &space_id) {
            for m in space.members.iter().filter(|m| !m.kicked) {
                let name = m.display_name.as_deref().unwrap_or(&m.pubkey);
                if q.is_empty() || name.to_lowercase().contains(&q) {
                    results.push(MentionSuggestion {
                        kind: "member".into(),
                        id: m.pubkey.clone(),
                        label: name.to_string(),
                        sublabel: m.role.clone(),
                        avatar_b64: m.avatar_blob.as_ref().map(|b| {
                            use base64::Engine;
                            base64::engine::general_purpose::STANDARD.encode(b)
                        }),
                        color_accent: m.color_accent.clone(),
                    });
                }
            }
        }
    }

    if kind == "all" || kind == "board" {
        let mut stmt = storage.conn().prepare(
            "SELECT board_id FROM space_boards WHERE space_id = ?1"
        ).map_err(|e| e.to_string())?;
        let board_ids: Vec<String> = stmt.query_map([&space_id], |row| row.get(0))
            .map_err(|e| e.to_string())?
            .filter_map(|r| r.ok())
            .filter(|id: &String| !id.ends_with("-chat"))
            .collect();
        for board_id in board_ids {
            if let Ok(doc) = storage.load_board(&board_id) {
                let title = kanban_core::board::get_board_title(&doc)
                    .unwrap_or_else(|_| board_id[..8.min(board_id.len())].to_string());
                if q.is_empty() || title.to_lowercase().contains(&q) {
                    results.push(MentionSuggestion {
                        kind: "board".into(),
                        id: board_id,
                        label: title,
                        sublabel: None,
                        avatar_b64: None,
                        color_accent: None,
                    });
                }
            }
        }
    }

    if kind == "all" || kind == "card" {
        let like = format!("%{q}%");
        let mut stmt = storage.conn().prepare(
            "SELECT card_id, board_id, title, column_name FROM card_search_index
             WHERE space_id = ?1 AND (title LIKE ?2 OR ?2 = '%%')
             LIMIT 20"
        ).map_err(|e| e.to_string())?;
        let rows: Vec<(String, String, String, String)> = stmt.query_map(
            rusqlite::params![space_id, like],
            |row| Ok((row.get(0)?, row.get(1)?, row.get(2)?, row.get(3)?))
        ).map_err(|e| e.to_string())?
        .filter_map(|r| r.ok())
        .collect();
        for (card_id, _board_id, title, column_name) in rows {
            results.push(MentionSuggestion {
                kind: "card".into(),
                id: card_id,
                label: title,
                sublabel: Some(column_name),
                avatar_b64: None,
                color_accent: None,
            });
        }
    }

    Ok(results)
}
```

```javascript
// ── Chat Panel ─────────────────────────────────────────────────────────────
let _chatSpaceId = null;
let _chatPollInterval = null;
let _chatLastTs = null;
let _chatUnreadCount = 0;
let _chatAllMembers = {};
let _autocompleteItems = [];
let _autocompleteIdx = -1;
let _mentionStart = -1;

async function openChatPanel(spaceId, spaceName) {
  _chatSpaceId = spaceId;
  document.getElementById('chat-panel-title').textContent = '💬 ' + (spaceName || 'Chat');
  document.getElementById('chat-panel').classList.add('open');
  _chatUnreadCount = 0;
  updateUnreadBadge();
  await loadChatMessages();
  _chatPollInterval = setInterval(loadChatMessages, 3000);
}

function closeChatPanel() {
  document.getElementById('chat-panel').classList.remove('open');
  if (_chatPollInterval) { clearInterval(_chatPollInterval); _chatPollInterval = null; }
}

function updateUnreadBadge() {
  const badge = document.getElementById('chat-unread-badge');
  if (_chatUnreadCount > 0 && !document.getElementById('chat-panel').classList.contains('open')) {
    badge.textContent = _chatUnreadCount > 9 ? '9+' : _chatUnreadCount;
    badge.classList.add('visible');
  } else {
    badge.classList.remove('visible');
  }
}

async function loadChatMessages() {
  if (!_chatSpaceId) return;
  try {
    const msgs = await invoke('get_chat_messages_cmd', {
      spaceId: _chatSpaceId, limit: 50, beforeTs: null
    });
    // Update member presence dots in header
    const seen = {};
    const dotsEl = document.getElementById('chat-member-dots');
    dotsEl.innerHTML = '';
    msgs.forEach(m => {
      if (!seen[m.author]) {
        seen[m.author] = true;
        const dot = document.createElement('div');
        dot.className = 'chat-member-dot';
        const color = m.color_accent || '#5a6a9a';
        if (m.avatar_b64) {
          dot.style.cssText = `background-image:url(data:image/png;base64,${m.avatar_b64});background-size:cover;`;
        } else {
          dot.style.background = color;
          dot.textContent = (m.display_name || '?')[0].toUpperCase();
        }
        dot.title = m.display_name || m.author.slice(0, 8);
        dotsEl.appendChild(dot);
      }
    });

    // Check for new messages
    const latestTs = msgs.length > 0 ? msgs[0].created_at : 0;
    if (_chatLastTs !== null && latestTs > _chatLastTs && !document.getElementById('chat-panel').classList.contains('open')) {
      _chatUnreadCount++;
      updateUnreadBadge();
    }
    _chatLastTs = latestTs;

    renderChatMessages(msgs);
  } catch (_) {}
}

function renderChatMessages(msgs) {
  const container = document.getElementById('chat-messages');
  const emptyEl = document.getElementById('chat-empty-msg');
  if (msgs.length === 0) {
    emptyEl.style.display = '';
    // Clear all but empty msg
    Array.from(container.children).forEach(c => { if (c !== emptyEl) c.remove(); });
    return;
  }
  emptyEl.style.display = 'none';
  // Re-render all (simple approach for now)
  Array.from(container.children).forEach(c => { if (c !== emptyEl) c.remove(); });
  const myPubkey = window._myPubkey || '';
  // msgs is newest-first; render oldest-first
  const ordered = [...msgs].reverse();
  ordered.forEach(m => {
    const div = document.createElement('div');
    div.className = 'chat-msg';
    const color = m.color_accent || '#5a6a9a';
    const initial = (m.display_name || '?')[0].toUpperCase();
    const avatarStyle = m.avatar_b64
      ? `background-image:url(data:image/png;base64,${m.avatar_b64});background-size:cover;`
      : `background:${color};`;
    const nameColor = m.color_accent || '#aaa';
    const ts = formatTs(m.created_at);
    // Render text with refs highlighted
    const textHtml = renderChatText(m.text, m.refs);
    div.innerHTML = `
      <div class="chat-msg-avatar" style="${avatarStyle}">${m.avatar_b64 ? '' : initial}</div>
      <div class="chat-msg-body">
        <div class="chat-msg-meta">
          <span class="chat-msg-name" style="color:${nameColor};">${m.display_name || m.author.slice(0,8)}</span>
          <span class="chat-msg-ts">${ts}</span>
        </div>
        <div class="chat-msg-text">${textHtml}</div>
      </div>`;
    container.appendChild(div);
  });
  container.scrollTop = container.scrollHeight;
}

function renderChatText(text, refs) {
  let html = text.replace(/</g, '&lt;').replace(/>/g, '&gt;');
  // Highlight @mentions and #refs as chips
  html = html.replace(/@(\S+)/g, (_, name) =>
    `<span class="chat-mention-chip" style="color:#aaa;">@${name}</span>`
  );
  html = html.replace(/#([\w\s-]+)/g, (match, label) => {
    const ref = refs.find(r => r.label === label.trim());
    if (ref) {
      return `<span class="chat-ref-chip" data-kind="${ref.kind}" data-id="${ref.id}">#${label}</span>`;
    }
    return `<span style="color:#666;">#${label}</span>`;
  });
  return html;
}

// Navigate to a #ref when clicked
document.addEventListener('click', (e) => {
  const chip = e.target.closest('.chat-ref-chip');
  if (!chip) return;
  const kind = chip.dataset.kind;
  const id = chip.dataset.id;
  if (kind === 'board') {
    // Open that board
    const spaceItem = document.querySelector(`.space-row.active`);
    if (spaceItem) openBoard(id);
  } else if (kind === 'card') {
    openCardById(id);
  }
});

// ── Autocomplete ──────────────────────────────────────────────────────────
const chatInput = document.getElementById('chat-input');
const autocompleteEl = document.getElementById('chat-autocomplete');

chatInput.addEventListener('input', async (e) => {
  const val = chatInput.value;
  const pos = chatInput.selectionStart;
  // Find last @ or # before cursor
  const before = val.slice(0, pos);
  const atMatch = before.match(/@(\w*)$/);
  const hashMatch = before.match(/#([\w\s]*)$/);
  if (atMatch) {
    _mentionStart = pos - atMatch[0].length;
    await showAutocomplete(atMatch[1], 'member');
  } else if (hashMatch) {
    _mentionStart = pos - hashMatch[0].length;
    await showAutocomplete(hashMatch[1], 'all');
  } else {
    hideAutocomplete();
  }
});

async function showAutocomplete(query, kind) {
  if (!_chatSpaceId) return;
  const kindParam = kind === 'member' ? 'member' : 'all';
  try {
    _autocompleteItems = await invoke('get_mention_suggestions_cmd', {
      spaceId: _chatSpaceId, query, kind: kindParam
    });
  } catch (_) { _autocompleteItems = []; }
  if (_autocompleteItems.length === 0) { hideAutocomplete(); return; }
  _autocompleteIdx = -1;
  renderAutocomplete();
  autocompleteEl.classList.add('open');
}

function hideAutocomplete() {
  autocompleteEl.classList.remove('open');
  _autocompleteItems = [];
  _autocompleteIdx = -1;
  _mentionStart = -1;
}

function renderAutocomplete() {
  const groups = { member: [], card: [], board: [] };
  _autocompleteItems.forEach((item, i) => groups[item.kind]?.push({ ...item, _idx: i }));
  let html = '';
  const labels = { member: 'People', card: 'Cards', board: 'Boards' };
  for (const [k, items] of Object.entries(groups)) {
    if (!items.length) continue;
    html += `<div class="autocomplete-group-label">${labels[k]}</div>`;
    items.forEach(item => {
      const color = item.color_accent || '#5a6a9a';
      const initial = item.label[0]?.toUpperCase() || '?';
      const avatarStyle = item.avatar_b64
        ? `background-image:url(data:image/png;base64,${item.avatar_b64});background-size:cover;`
        : `background:${color};`;
      const icon = k === 'board' ? '▣' : k === 'card' ? '□' : '';
      const sub = item.sublabel ? `<span class="autocomplete-item-sub">${item.sublabel}</span>` : '';
      html += `<div class="autocomplete-item" data-idx="${item._idx}">
        <span class="autocomplete-item-avatar" style="${avatarStyle}">${item.avatar_b64 ? '' : (icon || initial)}</span>
        <span class="autocomplete-item-label">${item.label}</span>${sub}</div>`;
    });
  }
  autocompleteEl.innerHTML = html;
  autocompleteEl.querySelectorAll('.autocomplete-item').forEach(el => {
    el.onclick = () => insertMention(parseInt(el.dataset.idx));
  });
}

function insertMention(idx) {
  const item = _autocompleteItems[idx];
  if (!item) return;
  const val = chatInput.value;
  const pos = chatInput.selectionStart;
  const prefix = item.kind === 'member' ? '@' : '#';
  const before = val.slice(0, _mentionStart);
  const after = val.slice(pos);
  chatInput.value = before + prefix + item.label + ' ' + after;
  // Store ref for send
  if (!window._pendingRefs) window._pendingRefs = [];
  window._pendingRefs.push({ kind: item.kind, id: item.id, label: item.label });
  hideAutocomplete();
  chatInput.focus();
}

chatInput.addEventListener('keydown', (e) => {
  if (autocompleteEl.classList.contains('open')) {
    if (e.key === 'ArrowDown') { e.preventDefault(); selectAutocomplete(1); return; }
    if (e.key === 'ArrowUp') { e.preventDefault(); selectAutocomplete(-1); return; }
    if (e.key === 'Enter' || e.key === 'Tab') {
      if (_autocompleteIdx >= 0) { e.preventDefault(); insertMention(_autocompleteIdx); return; }
    }
    if (e.key === 'Escape') { hideAutocomplete(); return; }
  }
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    sendChatMessage();
  }
});

function selectAutocomplete(dir) {
  const items = autocompleteEl.querySelectorAll('.autocomplete-item');
  if (!items.length) return;
  _autocompleteIdx = Math.max(0, Math.min(items.length - 1, _autocompleteIdx + dir));
  items.forEach((el, i) => el.classList.toggle('selected', i === _autocompleteIdx));
}

async function sendChatMessage() {
  const text = chatInput.value.trim();
  if (!text || !_chatSpaceId) return;
  const refs = window._pendingRefs || [];
  window._pendingRefs = [];
  chatInput.value = '';
  hideAutocomplete();
  try {
    await invoke('send_chat_message_cmd', {
      spaceId: _chatSpaceId, text, refs
    });
    await loadChatMessages();
  } catch (e) {
    chatInput.value = text;
    console.error('Chat send error:', e);
  }
}

document.getElementById('btn-close-chat').onclick = closeChatPanel;
document.getElementById('btn-open-chat').onclick = () => {
  const activeSpace = window._activeSpaceId;
  if (!activeSpace) return;
  const spaceName = document.querySelector('.space-row.active .space-row-name')?.textContent || 'Chat';
  openChatPanel(activeSpace, spaceName);
};
```
